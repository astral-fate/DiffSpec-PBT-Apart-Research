def pre(i):
    return isinstance(i, list) and all(isinstance(x, int) for x in i)

def post(i, o):
    if not isinstance(o, list):
        return False
    if not all(isinstance(x, int) for x in o):
        return False
    if not pre(i):
        return False
    uniq_i = set(i)
    uniq_o = set(o)
    return uniq_o == uniq_i and len(o) == len(uniq_i)
