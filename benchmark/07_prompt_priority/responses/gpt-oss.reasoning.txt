def pre(i):
    if not isinstance(i, tuple) or len(i)!=3: return False
    system, user, sources = i
    if not isinstance(system, str) or not isinstance(user, str): return False
    if not isinstance(sources, (list, tuple)): return False
    for s in sources:
        if not isinstance(s, str):
            return False
    return True

def post(i,o):
    if not pre(i): return False
    system, user, sources = i
    if not isinstance(o, list): return False
    if len(o) != 2 + len(sources): return False
    if o[0] != system or o[1] != user: return False
    if o[2:] != list(sources): return False
    # also ensure each element is string
    for elem in o:
        if not isinstance(elem, str):
            return False
    return True
