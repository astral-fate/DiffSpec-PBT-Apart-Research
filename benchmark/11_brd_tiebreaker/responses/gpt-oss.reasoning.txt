def pre(i):
    if not isinstance(i, list):
        return False
    for rec in i:
        if not (isinstance(rec, tuple) and len(rec) == 4):
            return False
        for field in rec:
            if not isinstance(field, int):
                return False
    return True

post(i,o):
    if not isinstance(o, list):
        return False
    if len(i) != len(o):
        return False
    from collections import Counter
    if Counter(i) != Counter(o):
        return False
    def key(r):
        return (-r[0], -r[1], -r[2], r[3])
    for a,b in zip(o, o[1:]):
        if key(a) > key(b):
            return False
    return True
